        public int rand(int x)
        {

            Random g = new Random();

            switch (Properties.Settings.Default.Level)
            {
                case 1:
                    x = g.Next(0, 9);
                    break;

                case 2:
                    x = g.Next(10, 19);
                    break;

                case 3:
                    x = g.Next(0, 29);
                    break;
            }
            return x;
        }
